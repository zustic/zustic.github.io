// store.ts
  
import { create } from 'zustic';
type CounterStore = {
  count: number;
  inc: () => void;
  dec: () => void;
  reset: () => void;
};


export const useCounter = create<CounterStore>((set) => ({
  count: 0,
  inc: () => set((state) => ({ count: state.count + 1 })),
  dec: () => set((state) => ({ count: state.count - 1 })),
  reset: () => set({ count: 0 }),
}));
